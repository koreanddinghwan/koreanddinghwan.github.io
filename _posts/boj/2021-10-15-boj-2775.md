def apart(a,b): #a층b호
    if a == 0:
        return b
    if b == 1:
        return 1

        
    if lst[a][b-1] == -1: #만약 a층 b호가 계산이 안되어있다면,
        lst[a][b-1] = apart(a,b-1) + apart(a-1,b)
        


    return lst[a][b-1]
